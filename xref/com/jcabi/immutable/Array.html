42  /**
43   * Array as an object.
44   *
45   * <p>This class is truly immutable. This means that it never changes
46   * its encapsulated values and is annotated with {@code &#64;Immutable}
47   * annotation.
48   *
49   * @param <T> Value key type
50   * @author Yegor Bugayenko (yegor@tpc2.com)
51   * @version $Id$
52   * @since 0.1
53   */
54  @Immutable
55  @Loggable(Loggable.DEBUG)
56  @SuppressWarnings({ "unchecked", "PMD.TooManyMethods" })
57  public final class Array<T> implements List<T> {
58  
59      /**
60       * All values.
61       */
62      @Immutable.Array
63      private final transient T[] values;
64  
65      /**
66       * Public ctor, for an zero-length empty array.
67       */
68      public Array() {
69          this((T[]) new Object[0]);
70      }
71  
72      /**
73       * Public ctor, from an array of values.
74       * @param list Items to encapsulate
75       */
76      public Array(final T... list) {
77          this.values = list;
78      }
79  
80      /**
81       * Public ctor.
82       * @param list Items to encapsulate
83       * @since 0.12
84       */
85      public Array(final Iterable<T> list) {
86          if (list == null) {
87              throw new IllegalArgumentException(
88                  "list of objects can't be NULL"
89              );
90          }
91          if (list instanceof Array) {
92              this.values = ((Array<T>) list).values;
93          } else if (list instanceof Collection) {
94              final Collection<T> col = Collection.class.cast(list);
95              this.values = (T[]) new Object[col.size()];
96              col.toArray(this.values);
97          } else {
98              final Collection<T> items = new LinkedList<T>();
99              for (final T item : list) {
100                 items.add(item);
101             }
102             this.values = (T[]) new Object[items.size()];
103             items.toArray(this.values);
104         }
105     }
106 
107     /**
108      * Make a new one with an extra entry, at the end of array (will be
109      * extended by one extra element).
110      * @param value The value
111      * @return New vector
112      */
113     public Array<T> with(final T value) {
114         if (value == null) {
115             throw new IllegalArgumentException(
116                 "argument of Array#with() can't be NULL"
117             );
118         }
119         final T[] items = (T[]) new Object[this.values.length + 1];
120         System.arraycopy(this.values, 0, items, 0, this.values.length);
121         items[this.values.length] = value;
122         return new Array<T>(items);
123     }
124 
125     /**
126      * Make a new extra entries, at the end of array.
127      * @param vals The values
128      * @return New vector
129      */
130     public Array<T> with(final Iterable<T> vals) {
131         if (vals == null) {
132             throw new IllegalArgumentException(
133                 "arguments of Array#with() can't be NULL"
134             );
135         }
136         final Array<T> array;
137         if (vals instanceof Collection) {
138             final T[] items = (T[]) new Object[
139                 this.values.length + Collection.class.cast(vals).size()
140             ];
141             System.arraycopy(this.values, 0, items, 0, this.values.length);
142             int idx = this.values.length;
143             for (final T value : vals) {
144                 items[idx] = value;
145                 ++idx;
146             }
147             array = new Array<T>(items);
148         } else {
149             final Collection<T> list = new LinkedList<T>();
150             list.addAll(Arrays.asList(this.values));
151             for (final T value : vals) {
152                 list.add(value);
153             }
154             array = new Array<T>(list);
155         }
156         return array;
157     }
158 
159     /**
160      * Make a new one with an extra entry at the given position.
161      * @param pos Position to replace
162      * @param value The value
163      * @return New array
164      */
165     public Array<T> with(final int pos, final T value) {
166         if (value == null) {
167             throw new IllegalArgumentException(
168                 "second argument of Array#with() can't be NULL"
169             );
170         }
171         final T[] temp = (T[]) new Object[
172             Math.max(this.values.length, pos + 1)
173         ];
174         System.arraycopy(this.values, 0, temp, 0, this.values.length);
175         temp[pos] = value;
176         return new Array<T>(temp);
177     }
178 
179     /**
180      * Make a new array, without this element.
181      *
182      * <p>The method throws {@link ArrayIndexOutOfBoundsException} if such
183      * position is absent in the array.
184      *
185      * @param idx The position to remove
186      * @return New array
187      */
188     public Array<T> without(final int idx) {
189         if (idx >= this.values.length) {
190             throw new ArrayIndexOutOfBoundsException(
191                 String.format(
192                     "index %d is out of bounds: [0..%d]",
193                     idx, this.values.length
194                 )
195             );
196         }
197         if (idx < 0) {
198             throw new ArrayIndexOutOfBoundsException(
199                 String.format("index can't be negative: %d", idx)
200             );
201         }
202         final T[] items = (T[]) new Object[this.values.length - 1];
203         System.arraycopy(this.values, 0, items, 0, idx);
204         System.arraycopy(
205             this.values, idx + 1, items, idx, this.values.length - idx - 1
206         );
207         return new Array<T>(items);
208     }
209 
210     /**
211      * Make a new array, without this element (or the same array if such
212      * an element is absent).
213      * @param item The element to remove
214      * @return New array
215      * @since 1.4
216      */
217     public Array<T> less(final T item) {
218         int idx = -1;
219         for (int pos = 0; pos < this.values.length; ++pos) {
220             if (this.values[pos].equals(item)) {
221                 idx = pos;
222                 break;
223             }
224         }
225         final Array<T> array;
226         if (idx >= 0) {
227             array = this.without(idx);
228         } else {
229             array = this;
230         }
231         return array;
232     }

234     @Override
235     public int hashCode() {
236         return Arrays.hashCode(this.values);
237     }
238 
239     @Override
240     public boolean equals(final Object object) {
241         return object instanceof Array
242             && Arrays.deepEquals(this.values, Array.class.cast(object).values);
243     }
244 
245     @Override
246     public String toString() {
247         final StringBuilder text = new StringBuilder(0);
248         for (final T item : this.values) {
249             if (text.length() > 0) {
250                 text.append(", ");
251             }
252             text.append(item);
253         }
254         return text.toString();
255     }

259         return this.values.length;

264         return this.values.length == 0;
265     }

272     @Override
273     public Iterator<T> iterator() {
274         return Collections.unmodifiableList(
275             Arrays.asList(this.values)
276         ).iterator();
277     }

279     @Override
280     public Object[] toArray() {
281         final Object[] array = new Object[this.values.length];
282         System.arraycopy(this.values, 0, array, 0, this.values.length);
283         return array;
284     }
285 
286     @Override
287     public <T> T[] toArray(final T[] array) {
288         final T[] target;
289         if (array.length == this.values.length) {
290             target = array;
291         } else {
292             target = (T[]) new Object[this.values.length];
293         }
294         System.arraycopy(this.values, 0, target, 0, this.values.length);
295         return target;
296     }

352     @Override
353     public T get(final int index) {
354         if (index < 0 || index >= this.values.length) {
355             throw new IndexOutOfBoundsException(
356                 String.format(
357                     "index %d is out of bounds, length=%d",
358                     index,
359                     this.values.length
360                 )
361             );
362         }
363         return this.values[index];
364     }
