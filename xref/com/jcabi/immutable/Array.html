42  /**
43   * Array as an object.
44   *
45   * <p>This class is truly immutable. This means that it never changes
46   * its encapsulated values and is annotated with {@code @Immutable}
47   * annotation.
48   * <p>
49   * Limitation:
50   * Encapsulated objects with exposed mutators can mutate their state.
51   *
52   * @param <T> Value key type
53   * @since 0.1
54   * @checkstyle MissingDeprecatedCheck (430 lines)
55   */
56  @Immutable
57  @Loggable(Loggable.DEBUG)
58  @SuppressWarnings({
59      "unchecked", "PMD.TooManyMethods",
60      "PMD.ConstructorOnlyInitializesOrCallOtherConstructors",
61      "PMD.OnlyOneConstructorShouldDoInitialization"
62  })
63  public final class Array<T> implements List<T> {
64  
65      /**
66       * All values.
67       */
68      @Immutable.Array
69      private final transient T[] values;
70  
71      /**
72       * Public ctor, for an zero-length empty array.
73       */
74      public Array() {
75          this((T[]) new Object[0]);
76      }
77  
78      /**
79       * Public ctor, from an array of values.
80       * @param list Items to encapsulate
81       */
82      public Array(final T... list) {
83          this.values = (T[]) new Object[list.length];
84          System.arraycopy(list, 0, this.values, 0, list.length);
85      }
86  
87      /**
88       * Public ctor.
89       * @param list Items to encapsulate
90       * @since 0.12
91       */
92      public Array(final Iterable<T> list) {
93          this.throwIfArgumentIsNull(list, "list of objects can't be NULL");
94          final Collection<T> items = new LinkedList<>();
95          for (final T item : list) {
96              items.add(item);
97          }
98          this.values = (T[]) new Object[items.size()];
99          items.toArray(this.values);
100     }
101 
102     /**
103      * Public ctor, from a generic array.
104      * @param array Array with items to encapsulate
105      */
106     public Array(final Array<T> array) {
107         this.values = array.values.clone();
108     }
109 
110     /**
111      * Public ctor, from a generic collection.
112      * @param collection Collection with items to encapsulate
113      */
114     public Array(final Collection<T> collection) {
115         this.values = (T[]) new Object[collection.size()];
116         collection.toArray(this.values);
117     }
118 
119     /**
120      * Make a new one with an extra entry, at the end of array (will be
121      * extended by one extra element).
122      * @param value The value
123      * @return New vector
124      */
125     public Array<T> with(final T value) {
126         this.throwIfArgumentIsNull(
127             value, "argument of Array#with() can't be NULL"
128         );
129         final T[] items = (T[]) new Object[this.values.length + 1];
130         System.arraycopy(this.values, 0, items, 0, this.values.length);
131         items[this.values.length] = value;
132         return new Array<>(items);
133     }
134 
135     /**
136      * Make a new extra entries, at the end of array.
137      * @param vals The values
138      * @return New vector
139      */
140     public Array<T> with(final Collection<T> vals) {
141         this.throwIfArgumentIsNull(
142             vals, "Collection argument of Array#with() can't be NULL"
143         );
144         final T[] items = (T[]) new Object[
145             this.values.length + Collection.class.cast(vals).size()
146         ];
147         System.arraycopy(this.values, 0, items, 0, this.values.length);
148         int idx = this.values.length;
149         for (final T value : vals) {
150             items[idx] = value;
151             ++idx;
152         }
153         return new Array<>(items);
154     }
155 
156     /**
157      * Make a new extra entries, at the end of array.
158      * @param vals The values
159      * @return New vector
160      */
161     public Array<T> with(final Iterable<T> vals) {
162         this.throwIfArgumentIsNull(
163             vals, "Iterable argument of Array#with() can't be NULL"
164         );
165         final Collection<T> list = new LinkedList<>();
166         list.addAll(Arrays.asList(this.values));
167         for (final T value : vals) {
168             list.add(value);
169         }
170         return new Array<>(list);
171     }
172 
173     /**
174      * Make a new one with an extra entry at the given position.
175      * @param pos Position to replace
176      * @param value The value
177      * @return New array
178      */
179     public Array<T> with(final int pos, final T value) {
180         this.throwIfArgumentIsNull(
181             value, "second argument of Array#with() can't be NULL"
182         );
183         final T[] temp = (T[]) new Object[
184             Math.max(this.values.length, pos + 1)
185         ];
186         System.arraycopy(this.values, 0, temp, 0, this.values.length);
187         temp[pos] = value;
188         return new Array<>(temp);
189     }
190 
191     /**
192      * Make a new array, without element on specific index.
193      *
194      * <p>The method throws {@link ArrayIndexOutOfBoundsException} if such
195      * position is absent in the array.
196      *
197      * @param idx The position to remove
198      * @return New array
199      */
200     public Array<T> withoutIndex(final int idx) {
201         if (idx >= this.values.length) {
202             throw new ArrayIndexOutOfBoundsException(
203                 String.format(
204                     "index %d is out of bounds: [0..%d]",
205                     idx, this.values.length
206                 )
207             );
208         }
209         if (idx < 0) {
210             throw new ArrayIndexOutOfBoundsException(
211                 String.format("index can't be negative: %d", idx)
212             );
213         }
214         final T[] items = (T[]) new Object[this.values.length - 1];
215         System.arraycopy(this.values, 0, items, 0, idx);
216         System.arraycopy(
217             this.values, idx + 1, items, idx, this.values.length - idx - 1
218         );
219         return new Array<>(items);
220     }
221 
222     /**
223      * Make a new array, without this element (or the same array if such
224      * an element is absent).
225      * @param item The element to remove
226      * @return New array
227      * @since 1.4
228      */
229     public Array<T> without(final T item) {
230         int idx = -1;
231         for (int pos = 0; pos < this.values.length; ++pos) {
232             if (this.values[pos].equals(item)) {
233                 idx = pos;
234                 break;
235             }
236         }
237         final Array<T> array;
238         if (idx >= 0) {
239             array = this.withoutIndex(idx);
240         } else {
241             array = this;
242         }
243         return array;
244     }

246     @Override
247     public int hashCode() {
248         return Arrays.hashCode(this.values);
249     }
250 
251     @Override
252     public boolean equals(final Object object) {
253         return object instanceof Array
254             && Arrays.deepEquals(this.values, Array.class.cast(object).values);
255     }
256 
257     @Override
258     public String toString() {
259         final StringBuilder text = new StringBuilder(0);
260         for (final T item : this.values) {
261             if (text.length() > 0) {
262                 text.append(", ");
263             }
264             text.append(item);
265         }
266         return text.toString();
267     }

269     @Override
270     public int size() {
271         return this.values.length;
272     }
273 
274     @Override
275     public boolean isEmpty() {
276         return this.values.length == 0;
277     }
278 
279     @Override
280     public boolean contains(final Object key) {
281         return Arrays.asList(this.values).contains(key);
282     }

284     @Override
285     public Iterator<T> iterator() {
286         return Collections.unmodifiableList(
287             Arrays.asList(this.values)
288         ).iterator();
289     }
290 
291     @Override
292     public Object[] toArray() {
293         final Object[] array = new Object[this.values.length];
294         System.arraycopy(this.values, 0, array, 0, this.values.length);
295         return array;
296     }
297 
298     @Override
299     public <T> T[] toArray(final T[] array) {
300         final T[] target;
301         if (array.length == this.values.length) {
302             target = array;
303         } else {
304             target = (T[]) new Object[this.values.length];
305         }
306         System.arraycopy(this.values, 0, target, 0, this.values.length);
307         return target;
308     }

371     @Override
372     public T get(final int index) {
373         if (index < 0 || index >= this.values.length) {
374             throw new IndexOutOfBoundsException(
375                 String.format(
376                     "index %d is out of bounds, length=%d",
377                     index,
378                     this.values.length
379                 )
380             );
381         }
382         return this.values[index];
383     }

440     /**
441      * Throws IllegalArgumentException if the input parameter is null.
442      * @param obj Object to check its nullity
443      * @param message Message to send with the exception
444      */

446         if (obj == null) {
447             throw new IllegalArgumentException(message);
